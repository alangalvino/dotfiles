:END:
#+begin_src ~/.config/xfce4/bin/toggle-panel
#!/bin/bash

if [ `ps aux | grep ' xfce4-panel '|  grep -v grep` ]; then
  killall xfce4-panel
else
  xfce4-panel --disable-wm-check
fi
#+end_src
